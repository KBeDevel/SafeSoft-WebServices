Codificación de cadenas texto/bytes a base64

Wiki de Base64

C#:


    // Encode

    public static string Base64Encode(string plainText) {

        var plainTextBytes = System.Text.Encoding.UTF8.GetBytes(plainText);

        return System.Convert.ToBase64String(plainTextBytes);
    }

    // Decode

    public static string Base64Decode(string base64EncodedData) {

        var base64EncodedBytes = System.Convert.FromBase64String(base64EncodedData);

        return System.Text.Encoding.UTF8.GetString(base64EncodedBytes);
    }

Java:


    // Si estás usando Java < 1.8
    import org.apache.commons.codec.binary.Base64;

    // Si estás usando Java >= 1.8
    import java.util.Base64;

    // Encode

    byte[] bytesEncoded = Base64.encodeBase64(str.getBytes());

    System.out.println("Encoded value: " + new String(bytesEncoded));
    
    // Decode

    byte[] valueDecoded = Base64.decodeBase64(bytesEncoded);

    System.out.println("Decoded value: " + new String(valueDecoded));

JavaScript / TypeScript:


    // Encode

    var rawString = "something"

    var encString = btoa(rawString)

    console.log("Encoded value: " + encString)

    // Decode

    var encString = "c29tZXRoaW5n"

    var decString = atob(encString)

    console.log("Decoded value: " + decString)

PHP


    # Encode

    $rawString = "something";

    $encString = base64_encode($rawString);

    echo "Encoded value: " . $encString;

    # Decode

    $encString = "c29tZXRoaW5n"

    $decString = base64_decode($encString);

    echo "Decoded value: " . $decString;
